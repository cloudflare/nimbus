title: Banner
tagline: "Announcement strip — semantic types, sanitised HTML content, optional persistent dismiss."
props:
  - name: content
    type: string
    required: true
    description: HTML string (sanitised) rendered as the banner body.
  - name: type
    type: '"note" | "tip" | "caution" | "danger"'
    defaultValue: '"note"'
    description: Visual treatment that conveys intent.
  - name: dismissible
    type: "{ id: string; days?: number }"
    description: Show a close button; persists dismissal to localStorage under the given id for the given number of days.

## Examples

### Dismissible

Set `dismissible` to add a close button. The dismissal persists across page loads.

<Example code={`<Banner
  type="caution"
  content="Maintenance window: API reads are read-only from 18:00 UTC."
  dismissible={{ id: "maint-2026-06", days: 1 }}
/>`} contentClass="w-full">
  <div class="w-full">
    <Banner
      type="caution"
      content="Maintenance window: API reads are read-only from 18:00 UTC."
      dismissible={{ id: "demo-banner", days: 1 }}
    />
  </div>
</Example>
